import sqlite3
import os

if os.path.exists("users.db"):
    os.remove("users.db")

conn = sqlite3.connect("users.db")
c = conn.cursor()

c.execute("""
CREATE TABLE users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    usuario TEXT UNIQUE,
    senha TEXT
)
""")

c.execute("""
CREATE TABLE historico (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    usuario TEXT,
    expressao TEXT,
    resultado REAL
)
""")

conn.commit()
conn.close()

print("BANCO RESETADO COM SUCESSO")
